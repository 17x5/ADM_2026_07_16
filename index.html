/** * Die optimierte render-Funktion: * Lädt alle Fazit-Elemente nacheinander, um 429 (Rate Limit) * und 503 (Server Überlastung) Fehler zu vermeiden. */ async function render() { console.log("Starte sequenzielles Laden..."); // 1. Definition der Aufgaben, die nacheinander ausgeführt werden sollen const tasks = [ { name: "Fazit 1", fn: generateKachelFazit1 }, { name: "Fazit 2", fn: generateKachelFazit2 }, { name: "Fazit 3", fn: generateKachelFazit3 }, { name: "Fazit 4", fn: generateKachelFazit4 }, { name: "Hauptfazit", fn: generateHauptFazit } ]; // 2. Schleife, die jede Funktion einzeln abwartet for (const task of tasks) { try { console.log(`Lade ${task.name}...`); await task.fn(); // Wartet, bis die API fertig ist // 3. Kurze Pause von 1 Sekunde (1000ms), um Google API zu schonen await new Promise(resolve => setTimeout(resolve, 1000)); } catch (error) { console.error(`Fehler beim Laden von ${task.name}:`, error); } } console.log("Alle Fazits erfolgreich geladen."); }
